<script>
    const teams = [
        { name: "Team A", total: 0, fairplay: 0 },
        { name: "Team B", total: 0, fairplay: 0 },
        { name: "Team C", total: 0, fairplay: 0 },
        { name: "Team D", total: 0, fairplay: 0 }
    ];

    function renderTable() {
        teams.sort((a, b) => b.total - a.total);

        const tbody = document.querySelector("#leaderboard tbody");
        tbody.innerHTML = "";

        teams.forEach(team => {
            const row = document.createElement("tr");
            row.innerHTML = `
                <td>${team.name}</td>
                <td>${team.total}</td>
                <td>${team.fairplay}</td>
            `;
            tbody.appendChild(row);
        });
    }

    renderTable();
</script>
